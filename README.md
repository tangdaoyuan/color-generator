# color-generator

> Generate Color By HSL

## Install
```bash
npm install @suger-tdy/color-generator
```

## Usage
```ts
import { pickRandomColor } from '@suger-tdy/color-generator'

const colors = pickRandomColor(5)
console.log(colors)

// [
//   {
//     "b": 13,
//     "g": 255,
//     "h": 123,
//     "r": 0,
//   },
//   {
//     "b": 149,
//     "g": 255,
//     "h": 155,
//     "r": 0,
//   },
//   {
//     "b": 0,
//     "g": 128,
//     "h": 30,
//     "r": 255,
//   },
// ]
```
